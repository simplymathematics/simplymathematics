 if(!cmdComponents[1]){
             window.open(/presidents);
         }else{
                Plotly.d3.csv('data/presidents.csv', function(rows) {
                Build list of unique presidents, heights, weights
                let presidents = [];
                let heights    = [];
                let weights    = [];
                rows.forEach(function(d) {
                 if (!presidents.includes(d['Name'])) {
                   presidents.push(d['Name']);
                   heights.push(d['Height']);
                   weights.push(d['Weight']);
                 }
                }

                 let president = cmdComponents[1]
                 let index = presidents.indexOf(president)
                 result = [heights[index], weights[index]]
                 this.type(result.join(), this.unlock.bind(this));
                }
